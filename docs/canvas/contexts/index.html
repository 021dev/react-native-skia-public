Contexts

React Native Skia is using its own React renderer. Currently, it is not possible to automatically share a React context between two renderers. This means that a React Native context won't be available from your drawing directly. We recommend that you prepare the data needed for your drawing outside the <Canvas> element. However, if you need to use a React context within your drawing, you will need to re-inject it. This module provides the useContextBridge hook from pmndrs/drei to help you bridge between contexts.

import {useTheme, ThemeProvider} from "./docs/getting-started/Theme";

  const { primary } = useTheme();

  const theme = useTheme();
  return (
    <Canvas style={{ flex: 1 }}>
      {/* We need to re-inject the context provider here.  */}
      <ThemeProvider primary={theme.primary}>
        <MyDrawing />
      </ThemeProvider>
    </Canvas>
  );
};

  return (
    <ThemeProvider primary="red">

    </ThemeProvider>
  );
};

Below is the context definition that was used in this example:

tsx
import type { ReactNode } from "react";
import React, { useContext, createContext } from "react";
Â 
interface Theme {
  primary: string;
}
Â 
export const ThemeContext = createContext<Theme | null>(null);
Â 
export const ThemeProvider = ({
  primary,
  children,
}: {
  primary: string;
  children: ReactNode;
}) => (
  <ThemeContext.Provider value={{ primary }}>
    {children}
  </ThemeContext.Provider>
);
Â 
export const useTheme = () => {
  const theme = useContext(ThemeContext);
  if (theme === null) {
    throw new Error("Theme provider not found");
  }
  return theme;
};
